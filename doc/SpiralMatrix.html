def initialize(arg=nil)
  self.matrix = case arg
                when Integer
                  SpiralMatrix.random_matrix(arg.abs )
                when Array
                  arg
                else
                  SpiralMatrix.random_matrix(DEFAULT_SIZE )
                end

def self.random_matrix(size, max=DEFAULT_MAX_RANDOM)
  Array.new(size.times.map { size.times.map { rand(max) }})
end

    each(index=nil) { |e| ... } click to toggle source

    Enumerating from the outer side of the matrix to the center There is reverse_each method that returns values from the center First parameter can receive index to use elements till this index

    # File matrix.rb, line 54
def each(index=nil, &block)
  index = index || spiral_output.length
  return spiral_output.to_enum unless block
  spiral_output[0..index].each { |e| yield e }
end

def enumerator
  @enumerator ||= Enumerator.new(spiral_output.length) { |y| spiral_output.reverse_each { |e| y << e }}
end
